-- SQLite database export
PRAGMA foreign_keys = ON;

BEGIN TRANSACTION;
CREATE TABLE IF NOT EXISTS "Student" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "user_id" INTEGER NOT NULL UNIQUE,
    "branch_id" INTEGER,
    "usn" TEXT NOT NULL UNIQUE,
    FOREIGN KEY("branch_id") REFERENCES "Branch"("id"),
    FOREIGN KEY("user_id") REFERENCES "User"("id")
);

-- Indexes
CREATE INDEX IF NOT EXISTS "Student_index_2"
ON "Student" ("usn");

CREATE TABLE IF NOT EXISTS "Faculty" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "user_id" INTEGER NOT NULL UNIQUE,
    "employee_id" TEXT NOT NULL UNIQUE,
    "dept_id" INTEGER,
    FOREIGN KEY("dept_id") REFERENCES "Department"("id"),
    FOREIGN KEY("user_id") REFERENCES "User"("id")
);

-- Indexes
CREATE INDEX IF NOT EXISTS "Faculty_index_2"
ON "Faculty" ("employee_id");

CREATE TABLE IF NOT EXISTS "Marks" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "student_id" INTEGER NOT NULL,
    "exam_id" INTEGER NOT NULL,
    "score" INTEGER NOT NULL,
    FOREIGN KEY("exam_id") REFERENCES "Exam"("id"),
    FOREIGN KEY("student_id") REFERENCES "Student"("id")
);


CREATE TABLE IF NOT EXISTS "Student_Subject" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "student_id" INTEGER NOT NULL,
    "subject_id" INTEGER NOT NULL,
    FOREIGN KEY("student_id") REFERENCES "Student"("id"),
    FOREIGN KEY("subject_id") REFERENCES "Subject"("id")
);


CREATE TABLE IF NOT EXISTS "Admin" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "user_id" INTEGER NOT NULL UNIQUE,
    "dept_id" INTEGER,
    FOREIGN KEY("user_id") REFERENCES "User"("id")
);


CREATE TABLE IF NOT EXISTS "Exam" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "type" TEXT NOT NULL,
    "subject_id" INTEGER NOT NULL,
    "max_marks" INTEGER NOT NULL,
    "date" DATETIME NOT NULL,
    FOREIGN KEY("subject_id") REFERENCES "Subject"("id")
);


CREATE TABLE IF NOT EXISTS "Attendance" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "student_id" INTEGER NOT NULL,
    "class_id" INTEGER NOT NULL,
    "status" BOOLEAN NOT NULL,
    "date" DATETIME NOT NULL,
    FOREIGN KEY("class_id") REFERENCES "Class"("id"),
    FOREIGN KEY("student_id") REFERENCES "Student"("id")
);


CREATE TABLE IF NOT EXISTS "User" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "name" TEXT NOT NULL,
    "email" TEXT NOT NULL UNIQUE,
    "role" TEXT NOT NULL,
    "uid" TEXT NOT NULL UNIQUE,
    "password" TEXT NOT NULL,
    "phone_no" TEXT,
    "dob" TEXT,
    "address" TEXT
);

-- Indexes
CREATE INDEX IF NOT EXISTS "User_index_2"
ON "User" ("uid");
CREATE INDEX IF NOT EXISTS "User_index_3"
ON "User" ("name");

CREATE TABLE IF NOT EXISTS "Department" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "name" TEXT NOT NULL UNIQUE,
    "dept_uid" TEXT NOT NULL UNIQUE
);


CREATE TABLE IF NOT EXISTS "Branch" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "name" TEXT NOT NULL UNIQUE,
    "branch_uid" TEXT NOT NULL UNIQUE
);


CREATE TABLE IF NOT EXISTS "Class" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "faculty_id" INTEGER NOT NULL,
    "subject_id" INTEGER NOT NULL,
    "date" DATE,
    "start_time" DATETIME,
    "end_time" DATETIME,
    FOREIGN KEY("faculty_id") REFERENCES "Faculty"("id"),
    FOREIGN KEY("subject_id") REFERENCES "Subject"("id")
);


CREATE TABLE IF NOT EXISTS "Faculty_Subject" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "faculty_id" INTEGER NOT NULL,
    "subject_id" INTEGER NOT NULL,
    FOREIGN KEY("faculty_id") REFERENCES "Faculty"("id"),
    FOREIGN KEY("subject_id") REFERENCES "Subject"("id")
);


CREATE TABLE IF NOT EXISTS "Subject" (
    "id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
    "name" TEXT NOT NULL UNIQUE,
    "code" TEXT NOT NULL UNIQUE
);

-- Indexes
CREATE INDEX IF NOT EXISTS "Subject_index_2"
ON "Subject" ("code");

COMMIT;
